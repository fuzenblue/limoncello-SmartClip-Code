/**
 * ============================================================================
 * FILE B10: CMakeLists.txt (Main Component)
 * ============================================================================
 * PURPOSE: Component-level CMake file that lists all source files.
 *
 * In ESP-IDF, each folder under main/ or components/ is a "component".
 * This CMakeLists.txt tells the build system which .c files to compile
 * and which include directories to use.
 *
 * The idf_component_register() function is ESP-IDF specific — it:
 *   1. Compiles the listed SRCS files
 *   2. Adds INCLUDE_DIRS to the compiler's -I search path
 *   3. Links the listed REQUIRES components (ESP-IDF standard libraries)
 * ============================================================================
 */

idf_component_register(
    SRCS
        "main.c"
        "sensor_grid.c"
        "flicker_fft.c"
        "pressure_voc_calc.c"
        "audio_features.c"
        "feature_vector.c"
        "ble_transmit.c"
        "motion_gate.c"

    INCLUDE_DIRS
        "."                        # Current directory (for .h files)
        "../include"               # model_data.h (generated by Python pipeline)

    REQUIRES
        driver                     # I2C, I2S, ADC, GPIO drivers
        esp_timer                  # High-resolution hardware timer
        esp_adc                    # ADC oneshot driver (v5.x)
        nvs_flash                  # Non-Volatile Storage (required for BLE)
        bt                         # Bluetooth (BLE) stack
        esp_dsp                    # Espressif DSP library (FFT, filtering)
        log                        # ESP logging framework
)
